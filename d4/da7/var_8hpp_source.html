497  template <typename S, require_convertible_t<S&, value_type>* = nullptr,
498  require_any_not_plain_type_t<T, S>* = nullptr>
499  var_value<T> operator=(const var_value<S>& other) {
500  arena_t<plain_type_t<T>> prev_val = vi_->val_;
501  vi_->val_ = other.val();
502  // no need to change any adjoints - these are just zeros before the reverse
503  // pass
504 
505  reverse_pass_callback(
506  [this_vi = this->vi_, other_vi = other.vi_, prev_val]() mutable {
507  this_vi->val_ = prev_val;
508 
509  // we have no way of detecting aliasing between this->vi_->adj_ and
510  // other.vi_->adj_, so we must copy adjoint before reseting to zero
511 
512  // we can reuse prev_val instead of allocating a new matrix
513  prev_val = this_vi->adj_;
514  this_vi->adj_.setZero();
515  other_vi->adj_ += prev_val;
516  });
517  return *this;
518  }
